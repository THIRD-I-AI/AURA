# Causal discovery extras — installed only by the causal_service pod and
# anything that needs gcm.attribute_anomalies / robust partial correlation.
#
#   pip install -r requirements.txt
#   pip install -r requirements-causal.txt
#
# Why these aren't in base:
#   * dowhy transitively pulls scipy + scikit-learn + statsmodels (~150MB
#     of C extensions). Every other AURA pod (api_gateway, scheduler,
#     uasr, etc.) was paying that install cost for code none of them ever
#     called.
#   * pingouin + statsmodels are needed for the robust partial-correlation
#     fallback (Moore-Penrose pseudoinverse handles collinearity among
#     control variables that the prior np.linalg.lstsq path silently
#     produced garbage for) and for the ADF stationarity guardrail.

# DoWhy graphical causal models — primary engine for
# gcm.attribute_anomalies in causal_service/discovery.py.
# Sprint 13 bumped the floor to 0.13 — older DoWhy used
# ``networkx.algorithms.d_separated`` which was removed in
# networkx 3.x, so any combination of (dowhy<0.13, networkx>=3.x)
# crashes on the first DoWhy model build. The de-skipped eval-gate
# caught this on the original CI run; bumping to 0.13 keeps the
# networkx range we have without forcing every other AURA pod onto
# a narrower one.
dowhy>=0.13,<0.16
networkx>=3.0,<4.0

# Robust partial correlation (collinearity-aware) + ADF stationarity test
# for the causal_service guardrail that refuses attribution when training
# data spans a regime change.
pingouin>=0.5.4,<1.0
statsmodels>=0.14,<1.0

# Sprint 9 — Counterfactual Audit Engine (Auditor tier)
# reportlab generates the printable PDF auditor report. Pure-Python with
# no system deps, ~5MB install. The PDF endpoint returns 503 if this
# package is missing rather than crashing the engine.
reportlab>=4.0,<5.0

# Sprint 12 — EconML LinearDRLearner replaces the linear-regression stub
# previously in the engine's ``double_ml`` slot. Doubly robust ATE with
# cross-fitted nuisance models (sklearn RandomForestClassifier /
# RandomForestRegressor) and statsmodels-derived closed-form CIs. The
# engine gracefully falls back to the DoWhy linear-regression stub when
# econml isn't importable, so this dep is *required* only inside the
# eval-gate CI job that asserts Layer 10 byte-identity against the real
# estimator. sklearn/scipy/statsmodels are already pulled by dowhy.
econml>=0.15,<0.17
